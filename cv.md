# Code example

const isPalindrome = (string) => {
  const formattedString = string.replaceAll(' ', '').toLowerCase();

  for (let i = 0; i < formattedString.length / 2; i++) {
    if (formattedString.at(i) !== formattedString.at(-i - 1)) {
      return false;
    }
  }

  return true;
};
